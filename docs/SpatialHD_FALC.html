SCT normalization

seurat_names <- c("obj1", "obj2")

sce_list <- list()  # holds the 2 separate results

for (nm in seurat_names) {
  message("Processing native Seurat object: ", nm)
  seu_obj <- get(nm)

  active_assay <- DefaultAssay(seu_obj)
  message("  Found active assay: '", active_assay, "'")

  seu_obj <- seu_obj[, colSums(GetAssayData(seu_obj, assay = active_assay, slot = "counts")) > 0]
  seu_obj <- SCTransform(seu_obj, assay = active_assay, verbose = FALSE)

  sce_final <- as.SingleCellExperiment(seu_obj)

  sct_residual_matrix <- matrix(
    0,
    nrow = nrow(sce_final),
    ncol = ncol(sce_final),
    dimnames = list(rownames(sce_final), colnames(sce_final))
  )
  computed_residuals <- seu_obj[["SCT"]]$scale.data
  matching_genes <- intersect(rownames(sct_residual_matrix), rownames(computed_residuals))
  sct_residual_matrix[matching_genes, ] <- computed_residuals[matching_genes, ]

  assays(sce_final)[["SCT_residuals"]] <- sct_residual_matrix

  hvg_genes <- VariableFeatures(seu_obj)
  rowData(sce_final)$hvg <- rownames(sce_final) %in% hvg_genes
  metadata(sce_final)$sct_hvgs <- VariableFeatures(seu_obj)
  sce_list[[nm]] <- sce_final   # keep them separate, indexed by name
  message("  Successfully stored normalized container for: ", nm)
}

hvg_plots <- list()

for (nm in c("obj1", "obj2")) {

  sce_obj <- sce_list[[nm]]
  message("Processing: ", nm)

  # --- sanity: logcounts here is SCT's 'data' slot, not a fresh lognorm ---
  if (!"logcounts" %in% assayNames(sce_obj)) {
    stop("No logcounts in ", nm, " - check what as.SingleCellExperiment() carried over.")
  }

  # --- HVGs from SCT, ranked by residual variance ---
  hvg_all <- rownames(sce_obj)[rowData(sce_obj)$hvg]
  res_var <- apply(assay(sce_obj, "SCT_residuals")[hvg_all, , drop = FALSE], 1, var)
  top_HVG <- names(sort(res_var, decreasing = TRUE))[1:20]
  message("  Top HVG: ", paste(head(top_HVG, 3), collapse = ", "), " ...")

  # --- spatial coordinates from barcodes ---
  barcodes <- colnames(sce_obj)
  bin_x <- as.numeric(gsub(".*_([0-9]+)_[0-9]+-1", "\\1", barcodes))
  bin_y <- as.numeric(gsub(".*_[0-9]+_([0-9]+)-1", "\\1", barcodes))
  if (anyNA(bin_x) || anyNA(bin_y)) {
    warning(sum(is.na(bin_x) | is.na(bin_y)), " barcodes in ", nm,
            " did not match the coordinate pattern.")
  }

  expr_mat <- logcounts(sce_obj)

  plot_list <- lapply(top_HVG, function(gene) {
    df <- data.frame(
      x = bin_x,
      y = bin_y,
      expression = as.numeric(expr_mat[gene, ])
    )
    ggplot(df, aes(x = x, y = y, color = expression)) +
      geom_point(size = 0.005, alpha = 0.5) +
      scale_color_viridis_c(option = "plasma") +
      coord_fixed() +
      theme_void() +
      labs(title = gene, color = "LogExpr") +
      theme(
        plot.title = element_text(hjust = 0.5, face = "bold", size = 10),
        legend.position = "none"
      )
  })

  combined_plot <- wrap_plots(plot_list, nrow = 4, ncol = 5)
  hvg_plots[[nm]] <- combined_plot
  print(combined_plot)

  message("Done: ", nm)
}

nm <- "obj2"
sce_obj <- sce_list[[nm]]
if (!"logcounts" %in% assayNames(sce_obj)) {
  sce_obj <- logNormCounts(sce_obj)
}

genes_of_interest <- c("CCL19", "CXCL13", "CCL11", "IL33","IL6","ICAM1","VCAM1", "PDPN","INFG","TNF","STAT3","STAT1","STAT4","CD4","CD8","FOXP3","IL17A","RORC")

barcodes <- colnames(sce_obj)
bin_x <- as.numeric(gsub(".*_([0-9]+)_[0-9]+-1", "\\1", barcodes))
bin_y <- as.numeric(gsub(".*_[0-9]+_([0-9]+)-1", "\\1", barcodes))

for (i in seq_along(genes_of_interest)) {
  gene <- genes_of_interest[i]

  # Decide if the gene is detectable
  in_object   <- gene %in% rownames(sce_obj)
  expr        <- if (in_object) logcounts(sce_obj)[gene, ] else NULL
  detectable  <- in_object && any(expr > 0)

  message("Plotting ", i, "/", length(genes_of_interest), ": ", gene,
          if (!detectable) "  [not detectable]" else "")

  if (detectable) {
    df <- data.frame(x = bin_x, y = bin_y, expression = expr)
    df <- df[order(df$expression), ]          # positive spots on top
    p <- ggplot(df, aes(x = x, y = y, color = expression)) +
      geom_point(size = 0.3, alpha = 0.6) +
      scale_color_viridis_c(option = "plasma") +
      coord_fixed() + theme_void() +
      labs(title = gene, color = "LogExpr") +
      theme(plot.title = element_text(hjust = 0.5, face = "bold", size = 12))

  } else {
    # grey tissue outline + "not detectable" annotation
    df <- data.frame(x = bin_x, y = bin_y)
    reason <- if (!in_object) "not in dataset" else "all zero"
    p <- ggplot(df, aes(x = x, y = y)) +
      geom_point(size = 0.3, alpha = 0.4, color = "grey80") +
      coord_fixed() + theme_void() +
      labs(title = paste0(gene, " - not detectable")) +
      annotate("text", x = mean(range(bin_x, na.rm = TRUE)),
               y = mean(range(bin_y, na.rm = TRUE)),
               label = paste0("Not detectable\n(", reason, ")"),
               fontface = "italic", size = 4, color = "grey30") +
      theme(plot.title = element_text(hjust = 0.5, face = "bold", size = 12))
  }

  print(p)
}

score_signature <- function(sce, genes) {
  mat <- SummarizedExperiment::assay(sce, "logcounts")[genes, , drop = FALSE]
  mat <- as.matrix(base::as.matrix(mat))     # force a real base matrix
  storage.mode(mat) <- "double"

  mu <- base::rowMeans(mat)
  sdv <- sqrt(base::rowSums((mat - mu)^2) / (ncol(mat) - 1))
  sdv[sdv == 0 | is.na(sdv)] <- 1            # zero-variance genes -> score 0

  z <- (mat - mu) / sdv                      # recycles down columns, correct here
  base::colMeans(z)
}
